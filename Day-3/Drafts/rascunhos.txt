num = "189"

melhor = -1

# 1 8 9 numeros
# 0 1 2 indices

for i in range(len(num)):
    print(f"valor de i: {i}")
    print(f"numero atual: {num[i]}")
    for j in range(i + 1, len(num)):
        print(f"valor de j: {j}")
        print(f"numero atual: {num[j]}")
        valor = int(num[i] + num[j])
        if valor > melhor:
            melhor = valor
            print(f"combinação atual encontrada: {valor}")

print(f"Melhor combinação encontrada: {melhor}")
